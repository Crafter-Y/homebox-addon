#include <tunables/global>

profile homebox flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/s6-overlay/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Homebox data (persistent addon storage)
  /data/** rw,

  # Start new profile for the Homebox binary
  /usr/bin/homebox cx -> homebox_service,

  profile homebox_service flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Network access for the web UI
    network inet tcp,
    network inet6 tcp,

    signal (receive) peer=*_homebox,

    # Data directory (SQLite database, uploaded files, temp files)
    /data/** rw,

    # Binary itself
    /usr/bin/homebox rix,

    # Required system files
    /bin/sh rix,
    /etc/passwd r,
    /etc/ssl/certs/** r,
    /dev/tty rw,
    /proc/self/fd/ r,
  }
}
